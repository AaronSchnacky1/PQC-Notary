### 2. `src/main.rs`
```rust
use quantumvault::{keygen, sign, verify};
use std::env;
use std::fs;
use std::path::PathBuf;

fn main() {
    println!("üõ°Ô∏è  QuantumVault Notary v1.0 (std) ‚Äî Sign Offline");

    let args: Vec<String> = env::args().collect();
    if args.len() != 2 {
        eprintln!("Usage: notary-v1-std <file>");
        return;
    }

    let path = PathBuf::from(&args[1]);
    let data = match fs::read(&path) {
        Ok(d) => d,
        Err(e) => {
            eprintln!("Error reading file: {}", e);
            return;
        }
    };

    let kp = keygen();
    let sig = sign(&data, &kp.secret);

    let sig_path = path.with_extension("sig");
    if fs::write(&sig_path, hex::encode(&sig)).is_err() {
        eprintln!("Failed to write .sig");
        return;
    }

    let pk_path = path.with_extension("pub");
    if fs::write(&pk_path, hex::encode(&kp.public)).is_err() {
        eprintln!("Failed to write .pub");
        return;
    }

    if verify(&data, &sig, &kp.public) {
        println!("‚úÖ Signed ‚Üí {}.sig", path.display());
        println!("   Public key ‚Üí {}.pub", path.display());
        println!("   VERIFIED");
    } else {
        println!("‚ùå Verification failed");
    }
}
```
